331void A1::BuildRootSignature()
332{
333 CD3DX12_DESCRIPTOR_RANGE texTable;
334 texTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0);
335
336 // Root parameter can be a table, root descriptor or root constants.
337 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
338
339 // Perfomance TIP: Order from most frequent to least frequent.
340 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
341 slotRootParameter[1].InitAsConstantBufferView(0);
342 slotRootParameter[2].InitAsConstantBufferView(1);
343 slotRootParameter[3].InitAsConstantBufferView(2);
344
345 auto staticSamplers = GetStaticSamplers();
346
347 // A root signature is an array of root parameters.
348 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
349 (UINT)staticSamplers.size(), staticSamplers.data(),
350 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
351
352 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
353 ComPtr<ID3DBlob> serializedRootSig = nullptr;
354 ComPtr<ID3DBlob> errorBlob = nullptr;
355 HRESULT hr = D3D12SerializeRootSignature(
356 &rootSigDesc,
357 D3D_ROOT_SIGNATURE_VERSION_1,
358 serializedRootSig.GetAddressOf(),
359 errorBlob.GetAddressOf()
360 );
361
362 if (errorBlob != nullptr)
363 {
364 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
365 }
366 ThrowIfFailed(hr);
367
368 ThrowIfFailed(md3dDevice->CreateRootSignature(
369 0,
370 serializedRootSig->GetBufferPointer(),
371 serializedRootSig->GetBufferSize(),
372 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
373}
